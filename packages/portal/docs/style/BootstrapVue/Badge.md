Customised BootstrapVue badge component. AKA 'pill', 'chip' or 'tag'.

Docs: https://bootstrap-vue.org/docs/components/badge

Used by:
- LinkBadge
- [RelatedCategoryTags](/#/Components/Related?id=relatedcategorytags)

Variant "secondary": regular, pill<br>
*Used as default*
```jsx
import { thumbnails } from '../../../../styleguide/images';

  <b-container>
    <b-badge>badge</b-badge>
    <b-badge pill>
      pill
    </b-badge>
    <b-badge pill>
      pill
      <button class="close">x</button>
    </b-badge>
    <b-badge
      pill
      href="https://www.europeana.eu"
    >
      pill link
    </b-badge>
    <b-badge
      pill
      href="https://www.europeana.eu"
      class="img-chip"
    >
      <b-img
        :src="thumbnails[1]"
        rounded="circle"
        class="mr-2"
      />
      pill image link
    </b-badge>
  </b-container>
```
